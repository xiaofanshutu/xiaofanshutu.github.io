如何快速学习一个知识点

转义字符

// 在控制台，输入 tab 键，可以实现命令补全 
\t ：一个制表位，实现对齐的功能
\n ：换行符
\\ ：一个\
\" :一个"
\' ：一个' 
\r :一个回车
//转义字符练习
public class App {

    //编写一个 main 方法
    public static void main(String[] args) {
        //\t ：一个制表位，实现对齐的功能
        System.out.println("北京\t天津\t上海");

        // \n ：换行符
        System.out.println("jack\nsmith\nmary");

        // \\ ：一个\
        System.out.println("C:\\Windows\\System32\\cmd.exe");

        // \" :一个"
        System.out.println("你好呀，\"bambi\"");

        // \' ：一个'
        System.out.println("你好呀，\'bambi\'");

        // \r :一个回车
        System.out.println("你好呀\r明天");
        // 解读
        // 1. 输出 你好呀
        // 2. \r 表示回车
        System.out.println("你好呀\r明天");
    }
}
